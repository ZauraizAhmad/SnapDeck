You said:
can onclick work on a div directly?
ChatGPT said:

Yes 👍 — you can use onclick directly on a <div> just like you would on a button or link.

Example:

html
<div onclick="alert('Div clicked!')" 
     style="width:200px;height:100px;background:lightblue;cursor:pointer;">
  Click Me
</div>

Or, using JavaScript:

html
<div id="myDiv" style="width:200px;height:100px;background:lightcoral;cursor:pointer;">
  Click Me Too
</div>

<script>
  document.getElementById("myDiv").onclick = function() {
    alert("Div clicked via JS!");
  };
</script>

⚠️ Just remember:

  • <div> isn’t inherently clickable, so for accessibility it’s better to make it look and behave like a button (add role="button" and tabindex="0", so it can also be focused and activated with keyboard).

  • If it’s meant to act like a button, sometimes using an actual <button> is cleaner.

3. script.js → Add at the bottom:

js
/* STORY OVERLAY */
let currentStoryIndex = 0;
let stories = [];

function openStory(index) {
  const overlay = document.getElementById("story-overlay");
  const img = document.getElementById("story-image");

  currentStoryIndex = index;
  img.src = stories[currentStoryIndex].src;

  overlay.classList.remove("hide");
}

function closeStory() {
  document.getElementById("story-overlay").classList.add("hide");
}

function nextStory() {
  currentStoryIndex = (currentStoryIndex + 1) % stories.length;
  document.getElementById("story-image").src = stories[currentStoryIndex].src;
}

function prevStory() {
  currentStoryIndex =
    (currentStoryIndex - 1 + stories.length) % stories.length;
  document.getElementById("story-image").src = stories[currentStoryIndex].src;
}

window.addEventListener("DOMContentLoaded", () => {
  stories = Array.from(document.querySelectorAll(".deck img"));

  stories.forEach((img, index) => {
    img.parentElement.addEventListener("click", () => openStory(index));
  });

  document.querySelector(".story-left").addEventListener("click", prevStory);
  document.querySelector(".story-right").addEventListener("click", nextStory);

  // Close overlay if clicked outside image
  document.getElementById("story-overlay").addEventListener("click", (e) => {
    if (e.target.id === "story-overlay") closeStory();
  });

  // Close with ESC key
  document.addEventListener("keydown", (e) => {
    if (e.key === "Escape") closeStory();
  });
});

👉 With this setup:

  • Clicking any deck thumbnail opens it full screen.

  • You can click left/right arrows to switch between stories.

  • Click outside the image or press ESC to close.
